힙 정렬(Heap Sort)


힙 정렬은 완전 이진 트리를 기본으로 하는 힙(Heap) 자료구조를 기반으로한 정렬 방식이다.

완전 이진 트리는 삽입할 때 왼쪽부터 차례대로 추가하는 이진트리를 말한다.

힙에는 부모 노드의 값이 자식 노드의 값보다 항상 큰 최대 힙과 그 반대인 최소 힙이 존재한다.


힙 정렬은 힙 생성 알고리즘을 사용하는데, 특정한 노드의 두 자식 중에서 더 큰 자식과 자신의

위치를 바꾸는 알고리즘이다. 힙 정렬은 처음에 받아온 배열들의 값을 이진 트리에 삽입한다.

이진 트리에 값을 삽입하고 나면 힙 생성 알고리즘을 실행하게 되고, 제일 큰 값이

트리의 맨 위로 올라오게 된다. 그리고 그렇게 찾아진 제일 큰 값을 제일 아래쪽의 노드와 위치를

바꾸고,그 위치에 고정시킨다. 이러한 방법으로 힙 생성 알고리즘을 정렬이 될 때 까지 반복한다.


힙 정렬은 추가적인 메모리를 필요로 하지 않으면서 항상 O(NlogN)이라는 시간 복잡도를 가지며

가장 큰 값 몇개만 필요할 때에 정말 유용하다. 단점은 데이터의 순서가 바뀌는 불안정한

(Unstable)정렬 이라는 것과, 데이터의 상태에 따라 다른 정렬 방법에 비해 느려진다는 점이 있다.

#include <stdio.h>

void swap(int *a, int *b) 

{

  int temp = *a;

  *a = *b;

  *b = temp;

}

void heapify(int arr[], int n, int i) 

{

  int largest = i;

  int left = 2 * i + 1;

  int right = 2 * i + 2;

  if (left < n && arr[left] > arr[largest])

    largest = left;

  if (right < n && arr[right] > arr[largest])

    largest = right;

  if (largest != i) 

  {

    swap(&arr[i], &arr[largest]);

    heapify(arr, n, largest);

  }

}

void heapSort(int arr[], int n) 

{

  // Build max heap

  for (int i = n / 2 - 1; i >= 0; i--)

    heapify(arr, n, i);

  for (int i = n - 1; i >= 0; i--) 

  {

    swap(&arr[0], &arr[i]);

    heapify(arr, i, 0);

  }

}

void printArray(int arr[], int n) 

{

  for (int i = 0; i < n; ++i)

    printf("%d ", arr[i]);

  printf("\n");

}

int main() 

{

  int arr[] = {1, 12, 9, 5, 6, 10};

  int n = sizeof(arr) / sizeof(arr[0]);

  heapSort(arr, n);

  printf("Sorted array is \n");

  printArray(arr, n);

}

public class HeapSort 

{ 

    public void sort(int arr[]) 

    { 

        int n = arr.length;

        for (int i = n / 2 - 1; i >= 0; i--) 

            heapify(arr, n, i); 

        for (int i=n-1; i>0; i--) 

        { 

            int temp = arr[0]; 

            arr[0] = arr[i]; 

            arr[i] = temp; 

            heapify(arr, i, 0); 

        } 

    } 

    void heapify(int arr[], int n, int i) 

    { 

        int largest = i;

        int l = 2*i + 1;

        int r = 2*i + 2;

        if (l < n && arr[l] > arr[largest]) 

            largest = l; 

        if (r < n && arr[r] > arr[largest]) 

            largest = r; 

        if (largest != i) 

        { 

            int swap = arr[i]; 

            arr[i] = arr[largest]; 

            arr[largest] = swap; 

  

            // Recursively heapify the affected sub-tree 

            heapify(arr, n, largest); 

        } 

    }

def heap_sort(integers):

    integers_clone = list(integers)

    n = len(integers_clone)

    for i in range(n, -1, -1):

        max_heapify(integers_clone, i, n)

    n -= 1

    while n > 0:

        integers_clone[0], integers_clone[n] = integers_clone[n], integers_clone[0]

        max_heapify(integers_clone, 0, n)

        n -= 1

    return integers_clone

def max_heapify(arr, i, n):

    largest = i

    left_child = (i*2) + 1

    right_child = (i*2) + 2

    if left_child < n and arr[left_child] > arr[largest]:

        largest = left_child

    if right_child < n and arr[right_child] > arr[largest]:

        largest = right_child

    if largest != i:

        arr[i], arr[largest] = arr[largest], arr[i]

        max_heapify(arr, largest, n)
