칵테일 정렬(Cocktail Sort)


칵테일 정렬은 칵테일 쉐이커 정렬 또는 양방향 버블 정렬 이라고도 한다.

버블 정렬의 극단적인 비교 횟수를 줄이기 위해 버블 정렬을 조금 변형한 알고리즘이다.

우선 좌측부터 우측까지 비교작업을 진행한다. 교환할 자료가 있다면 교환하면서 우측 지점까지

진행한다. 마지막으로 교환이 이루어 진 곳을 우측 지점으로 설정하고 다시 왼쪽으로 비교작업을 진행한다.

이번에도 동일하게 마지막으로 교환이 이루어진 곳을 좌측 지점으로 설정하고 다시 오른쪽으로 이동한다.

좌측 지점과 우측 지점이 교차하지 않았다면 아직 정렬할 자료가 남아있다는 의미이므로 교차 할 때 까지 반복한다.


최상의 경우 O(N)의 시간복잡도를 가지고 , 최악의 경우 O(N²)의 시간복잡도를 가진다.

#include  <stdio.h>

void swap(int *x, int *y) 

{

    if(x == y)

        return;

    *x ^= *y;

    *y ^= *x;

    *x ^= *y;

}

void cocktailsort(int *a, size_t n) 

{

    while(1) 

    {

        char flag;

        int it, i;

        size_t start[2] = {1, n - 1},

        end[2] = {n, 0},

        inc[2] = {1, -1};

        for(it = 0; it < 2; ++it) 

        {

    flag = 1;

    for(i = start[it]; i != end[it]; i += inc[it])

        if(a[i - 1] > a[i]) 

                {

            swap(a + i - 1, a + i);

            flag = 0;

        }

        if(flag)

        return;

        }

    }

}

int main(void) 

{

    int a[] = { 5, -1, 101, -4, 0, 1, 8, 6, 2, 3 };

    int i;

    size_t n = sizeof(a)/sizeof(a[0]);

    printf("Original Array:\n");

    for (i = 0; i < n; i++)

    printf("%d%s", a[i], i == n - 1 ? "\n" : " ");  

    printf("\nSorted Array:\n");

    cocktailsort(a, n);

    for ( i = 0; i < n; ++i) printf("%d ", a[i]);

    return 0;

}

void CocktailSort(int a[], int n) 

{ 

    bool swapped = true; 

    int start = 0; 

    int end = n - 1; 

    while (swapped) 

    {

        swapped = false; 

        for (int i = start; i < end; ++i) 

        { 

            if (a[i] > a[i + 1]) 

            {

                swap(a[i], a[i + 1]); 

                swapped = true; 

            } 

        }

        if (!swapped) break; 

        swapped = false;

        --end; 

        for (int i = end - 1; i >= start; --i) 

        { 

            if (a[i] > a[i + 1]) 

            { 

                swap(a[i], a[i + 1]); 

                swapped = true; 

            } 

        } 

        ++start; 

    } 

}

public class CocktailSort 

{ 

    void cocktailSort(int a[]) 

    { 

        boolean swapped = true; 

        int start = 0; 

        int end = a.length;

        while (swapped == true) 

        {

            swapped = false;

            for (int i = start; i < end - 1; ++i) 

            { 

                if (a[i] > a[i + 1]) 

                { 

                    int temp = a[i]; 

                    a[i] = a[i + 1]; 

                    a[i + 1] = temp; 

                    swapped = true; 

                } 

            }

            if (swapped == false) 

                break;

            swapped = false;

            end = end - 1;

            for (int i = end - 1; i >= start; i--) 

            { 

                if (a[i] > a[i + 1]) 

                { 

                    int temp = a[i]; 

                    a[i] = a[i + 1]; 

                    a[i + 1] = temp; 

                    swapped = true; 

                } 

            }

            start = start + 1; 

        }
